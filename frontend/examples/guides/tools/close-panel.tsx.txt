import { usePillarTool, usePillar } from "@pillar-ai/react";

export function useTourTool() {
  const { close } = usePillar();

  usePillarTool({
    name: "start_tour",
    type: "trigger_tool",
    description: "Start the onboarding tour",
    execute: () => {
      close(); // Close panel first
      startOnboardingTour();
      return { success: true };
    },
  });
}
