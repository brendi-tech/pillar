<script setup lang="ts">
import { usePillar } from "@pillar-ai/vue";
import { watch } from "vue";

const { pillar } = usePillar();
const { user } = useAuth();

watch(
  () => user.value,
  (currentUser) => {
    if (currentUser && pillar.value) {
      pillar.value.identify(currentUser.id, {
        name: currentUser.name,
        email: currentUser.email,
        apiToken: currentUser.accessToken, // forwarded to API tool sources
      });
      pillar.value.setContext({
        userRole: currentUser.role,
        userState: currentUser.subscription?.status,
      });
    }
  },
  { immediate: true }
);
</script>
