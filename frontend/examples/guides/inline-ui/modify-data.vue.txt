<script setup lang="ts">
import { ref } from 'vue';
import type { ToolRenderProps } from '@pillar-ai/vue';

interface InviteData {
  emails: string[];
  role: string;
}

const props = defineProps<ToolRenderProps<InviteData>>();
const emails = ref([...props.data.emails]);

function removeEmail(email: string) {
  emails.value = emails.value.filter(e => e !== email);
}
</script>

<template>
  <div>
    <div v-for="email in emails" :key="email">
      {{ email }}
      <button @click="removeEmail(email)">×</button>
    </div>
    <button @click="props.onConfirm({ ...props.data, emails: emails })">Confirm</button>
  </div>
</template>
