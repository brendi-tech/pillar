usePillarTool({
  name: "delete_user",
  type: "trigger_tool",
  description: "Delete a user from the organization",
  requiredContext: { userRole: "admin" },
  inputSchema: {
    type: "object",
    properties: {
      userId: { type: "string", description: "ID of the user to delete" },
    },
    required: ["userId"],
  },
  execute: ({ userId }) => deleteUser(userId),
});
