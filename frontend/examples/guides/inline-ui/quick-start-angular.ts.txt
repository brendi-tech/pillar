import { Component, Input } from '@angular/core';
import { injectPillarTool, type ToolCardContext } from '@pillar-ai/angular';

interface InviteData {
  emails: string[];
  role: string;
}

@Component({
  selector: 'app-invite-card',
  standalone: true,
  template: `
    <div class="p-4 border rounded-lg">
      <h3 class="font-semibold mb-3">Invite Team Members</h3>
      <p class="text-sm text-gray-600 mb-4">
        {{ data.emails.length }} member{{ data.emails.length > 1 ? 's' : '' }} as {{ data.role }}
      </p>
      <ul class="mb-4 space-y-1">
        @for (email of data.emails; track email) {
          <li class="text-sm">{{ email }}</li>
        }
      </ul>
      <button
        (click)="handleSend()"
        [disabled]="!context.isReady"
        class="px-4 py-2 bg-blue-600 text-white rounded"
      >
        Send Invites
      </button>
    </div>
  `,
})
export class InviteCardComponent {
  @Input() data!: InviteData;
  @Input() sendResult!: (result: Record<string, unknown>) => Promise<void>;
  @Input() context!: ToolCardContext;

  async handleSend() {
    await inviteApi.send(this.data.emails, this.data.role);
    this.sendResult({ sent: this.data.emails.length, role: this.data.role });
  }
}

@Component({
  selector: 'app-invite-tools',
  standalone: true,
  template: '',
})
export class InviteToolsComponent {
  constructor() {
    injectPillarTool({
      name: 'invite_members',
      type: 'inline_ui',
      description: 'Invite team members to the workspace',
      inputSchema: {
        type: 'object',
        properties: {
          emails: { type: 'array', items: { type: 'string' }, description: 'Email addresses' },
          role: { type: 'string', enum: ['admin', 'member'], description: 'Role for new members' },
        },
        required: ['emails'],
      },
      render: InviteCardComponent,
    });
  }
}
