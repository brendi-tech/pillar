import { Pillar } from '@pillar-ai/sdk';

const pillar = await Pillar.init({ agentSlug: 'your-agent-slug' });

pillar.defineTool({
  name: 'invite_member',
  description: 'Invite a team member',
  type: 'trigger_tool',
  inputSchema: {
    type: 'object',
    properties: {
      email: { type: 'string', description: 'Email address to invite' },
      role: { type: 'string', description: 'Role: admin, member, or viewer' },
    },
    required: ['email', 'role'],
  },
  execute: ({ email, role }) => {
    openInviteModal({ email, role });
  },
});
