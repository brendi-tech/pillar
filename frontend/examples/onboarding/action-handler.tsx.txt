// app/layout.tsx (or your root component)
import { PillarProvider } from '@pillar-ai/react';
import { usePillarTools } from './hooks/usePillarTools';

function PillarToolsProvider({ children }) {
  // Register all tools - handlers are co-located in the hook
  usePillarTools();
  return children;
}

export default function RootLayout({ children }) {
  return (
    <PillarProvider productKey="your-product-key">
      <PillarToolsProvider>
        {children}
      </PillarToolsProvider>
    </PillarProvider>
  );
}
