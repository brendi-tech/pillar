import { usePillarTool } from "@pillar-ai/react";

// Define input type for type safety
interface InviteMemberInput {
  email: string;
  role: "admin" | "member" | "viewer";
}

export function useInviteTool() {
  usePillarTool({
    name: "invite_member",
    type: "trigger_tool",
    description: "Invite a new team member",
    inputSchema: {
      type: "object",
      properties: {
        email: { type: "string", description: "Email of the person to invite" },
        role: { type: "string", description: "Role: admin, member, or viewer" },
      },
      required: ["email", "role"],
    },
    execute: (input: InviteMemberInput) => {
      // TypeScript knows input has email and role
      console.log(`Inviting ${input.email} as ${input.role}`);
      sendInvite(input.email, input.role);
    },
  });
}
