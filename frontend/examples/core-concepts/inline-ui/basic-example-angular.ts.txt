import { Component } from '@angular/core';
import { injectPillarTool } from '@anthropic-ai/pillar-angular';
import { InviteMembersCardComponent } from './invite-members-card.component';

@Component({
  selector: 'app-invite-tools',
  standalone: true,
  template: '',
})
export class InviteToolsComponent {
  constructor() {
    injectPillarTool({
      name: 'invite_members',
      type: 'inline_ui',
      description: 'Invite team members to the workspace',
      parameters: {
        emails: { type: 'array', items: { type: 'string' } },
        role: { type: 'string', enum: ['admin', 'member'] },
      },
      execute: async ({ emails, role }) => ({ emails, role }),
      render: InviteMembersCardComponent,
    });
  }
}
