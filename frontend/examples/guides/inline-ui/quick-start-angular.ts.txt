import { Component, Input, Output, EventEmitter } from '@angular/core';
import { injectPillarTool } from '@anthropic-ai/pillar-angular';

interface InviteData {
  emails: string[];
  role: string;
}

@Component({
  selector: 'app-invite-card',
  standalone: true,
  template: `
    <div class="p-4 border rounded-lg">
      <h3 class="font-semibold mb-3">Invite Team Members</h3>
      <p class="mb-4">Inviting {{ data.emails.length }} members as {{ data.role }}</p>
      <div class="flex gap-2">
        <button (click)="onCancel.emit()">Cancel</button>
        <button (click)="onConfirm.emit()">Send Invites</button>
      </div>
    </div>
  `,
})
export class InviteCardComponent {
  @Input() data!: InviteData;
  @Output() onConfirm = new EventEmitter<Record<string, unknown>>();
  @Output() onCancel = new EventEmitter<void>();
}

@Component({
  selector: 'app-invite-tools',
  standalone: true,
  template: '',
})
export class InviteToolsComponent {
  constructor() {
    injectPillarTool({
      name: 'invite_members',
      type: 'inline_ui',
      description: 'Invite team members to the workspace',
      inputSchema: {
        type: 'object',
        properties: {
          emails: { type: 'array', items: { type: 'string' }, description: 'Email addresses' },
          role: { type: 'string', enum: ['admin', 'member'], description: 'Role for members' },
        },
        required: ['emails'],
      },
      render: InviteCardComponent,
    });
  }
}
