import { Component } from '@angular/core';
import { injectPillarTool } from '@pillar-ai/angular';

@Component({
  selector: 'app-source',
  standalone: true,
  template: ``,
})
export class SourceComponent {
  constructor() {
    injectPillarTool({
      name: 'add_source',
      type: 'trigger_tool',
      description: 'Add a new knowledge source',
      inputSchema: {
        url: { type: 'string', description: 'URL of the source to add' },
        name: { type: 'string', description: 'Display name for the source' },
      },
      execute: ({ url, name }) => {
        openAddSourceWizard({
          prefillUrl: url,
          prefillName: name,
        });
      },
    });
  }
}
