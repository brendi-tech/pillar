interface ThemeColors {
  /** Brand color for buttons, links, accents */
  primary?: string;
  
  /** Hover state for primary color */
  primaryHover?: string;
  
  /** Main background color */
  background?: string;
  
  /** Secondary background (cards, inputs) */
  backgroundSecondary?: string;
  
  /** Primary text color */
  text?: string;
  
  /** Secondary/muted text color */
  textMuted?: string;
  
  /** Border color */
  border?: string;
  
  /** Subtle border color */
  borderLight?: string;
}
