// app/layout.tsx (or your root component)
import { PillarProvider } from '@pillar-ai/react';
import { usePillarActions } from './hooks/usePillarActions';

function PillarActionsProvider({ children }) {
  // Register all actions - handlers are co-located in the hook
  usePillarActions();
  return children;
}

export default function RootLayout({ children }) {
  return (
    <PillarProvider productKey="your-product-key">
      <PillarActionsProvider>
        {children}
      </PillarActionsProvider>
    </PillarProvider>
  );
}
