import { Component, OnInit, inject } from '@angular/core';
import { injectPillar } from '@pillar-ai/angular';
import { AuthService } from './auth.service';

@Component({
  selector: 'app-user-context',
  standalone: true,
  template: ``,
})
export class UserContextComponent implements OnInit {
  private pillar = injectPillar();
  private auth = inject(AuthService);

  ngOnInit() {
    const user = this.auth.currentUser;
    if (user && this.pillar.pillar()) {
      this.pillar.pillar()!.identify(user.id, {
        name: user.name,
        email: user.email,
        apiToken: user.accessToken, // forwarded to API tool sources
      });
      this.pillar.pillar()!.setContext({
        userRole: user.role,
        userState: user.subscription?.status,
      });
    }
  }
}
