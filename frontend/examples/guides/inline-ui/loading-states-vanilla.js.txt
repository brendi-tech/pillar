render: (container, data, { onConfirm, onStateChange }) => {
  container.innerHTML = `<button id="confirm-btn">Confirm</button>`;

  container.querySelector('#confirm-btn').onclick = async () => {
    onStateChange?.('loading', 'Processing...');
    try {
      await doSomething(data.id);
      onStateChange?.('success', 'Done!');
      onConfirm();
    } catch (e) {
      onStateChange?.('error', 'Failed');
    }
  };

  return () => { container.innerHTML = ''; };
}
