render: (container, data, { onConfirm, onCancel }) => {
  // Use your existing CSS classes and design tokens
  container.innerHTML = `
    <div class="card">
      <div class="card-header">Confirm Action</div>
      <div class="card-content">
        <p>${data.message}</p>
        <div class="flex gap-2 mt-4">
          <button class="btn btn-outline" id="cancel-btn">Cancel</button>
          <button class="btn btn-primary" id="confirm-btn">Confirm</button>
        </div>
      </div>
    </div>
  `;

  container.querySelector('#cancel-btn').onclick = onCancel;
  container.querySelector('#confirm-btn').onclick = () => onConfirm();

  return () => { container.innerHTML = ''; };
}
