<script setup lang="ts">
import { usePillarTool, type ToolRenderProps } from '@pillar-ai/vue';
import { defineComponent } from 'vue';

interface DeleteData {
  itemId: string;
  itemName: string;
}

const ConfirmDeleteCard = defineComponent({
  props: ['data', 'onStateChange'],
  setup(props: ToolRenderProps<DeleteData>) {
    return {};
  },
  template: `
    <div class="p-4 border border-red-200 rounded-lg bg-red-50">
      <h3 class="font-semibold text-red-900">Delete {{ data.itemName }}?</h3>
      <p class="text-red-700 text-sm mt-1">This action cannot be undone.</p>
    </div>
  `,
});

usePillarTool({
  name: 'delete_item',
  type: 'inline_ui',
  description: 'Delete an item permanently',
  inputSchema: {
    type: 'object',
    properties: {
      itemId: { type: 'string' },
      itemName: { type: 'string' },
    },
    required: ['itemId', 'itemName'],
  },
  render: ConfirmDeleteCard,
});
</script>

<template><slot /></template>
