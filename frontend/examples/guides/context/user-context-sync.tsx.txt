function UserContextSync() {
  const { pillar } = usePillar();
  const { user } = useAuth();

  useEffect(() => {
    if (user && pillar) {
      pillar.identify(user.id, {
        name: user.name,
        email: user.email,
        apiToken: user.accessToken, // forwarded to API tool sources
      });
      pillar.setContext({
        userRole: user.role,
        userState: user.subscription?.status,
      });
    }
  }, [pillar, user]);

  return null;
}
