// hooks/usePillarTools.ts
import { usePillarTool } from '@pillar-ai/react';
import { useRouter } from 'next/navigation';

export function usePillarTools() {
  const router = useRouter();

  usePillarTool({
    name: 'open_settings',
    type: 'navigate',
    description: 'Navigate to the settings page',
    examples: ['open settings', 'go to settings', 'settings'],
    autoRun: true,
    execute: () => router.push('/settings'),
  });

  usePillarTool({
    name: 'contact_support',
    type: 'open_modal',
    description: 'Open contact support form',
    examples: ['contact support', 'talk to someone', 'get help'],
    execute: () => openSupportModal(),
  });
}
