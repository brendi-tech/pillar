import { usePillarTool } from '@pillar-ai/react';

usePillarTool({
  name: 'invite_members',
  type: 'inline_ui',
  description: 'Invite team members to the workspace',
  parameters: {
    emails: { type: 'array', items: { type: 'string' } },
    role: { type: 'string', enum: ['admin', 'member'] },
  },
  execute: async ({ emails, role }) => ({ emails, role }),
  render: ({ data, onConfirm, onCancel }) => (
    <InviteMembersCard
      emails={data.emails}
      role={data.role}
      onConfirm={onConfirm}
      onCancel={onCancel}
    />
  ),
});
