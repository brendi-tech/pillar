<script setup lang="ts">
import type { ToolRenderProps } from '@pillar-ai/vue';

interface MyData {
  id: string;
}

const props = defineProps<ToolRenderProps<MyData>>();

async function handleConfirm() {
  props.onStateChange?.('loading', 'Processing...');
  try {
    await doSomething(props.data.id);
    props.onStateChange?.('success', 'Done!');
    props.onConfirm();
  } catch (e) {
    props.onStateChange?.('error', 'Failed');
  }
}
</script>

<template>
  <button @click="handleConfirm">Confirm</button>
</template>
