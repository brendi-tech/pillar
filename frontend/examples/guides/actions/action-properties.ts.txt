  // Required
  name: string,             // Unique identifier for this tool
  description: string,      // AI uses this to match user intent
  execute: (input) => void, // Your execution logic

  // Optional
  type?: ToolType,          // navigate, trigger_action, query, inline_ui, external_link, copy_text

  examples?: string[],      // Example phrases that trigger this action
  inputSchema?: object,     // Schema for extracting data from user queries
  autoRun?: boolean,        // Execute without user clicking (default: false)
  autoComplete?: boolean,   // Mark done immediately (default: false)
  requiredContext?: object,  // Context required for tool to be available
});
