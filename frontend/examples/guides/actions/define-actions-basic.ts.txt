// src/hooks/usePillarTools.ts
import { usePillarTool } from "@pillar-ai/react";
import { useRouter } from "next/navigation";

export function usePillarTools() {
  const router = useRouter();

  usePillarTool({
    name: "open_settings",
    type: "navigate",
    description: "Navigate to the settings page",
    autoRun: true,
    execute: () => router.push("/settings"),
  });

  usePillarTool({
    name: "invite_member",
    type: "open_modal",
    description: "Open the invite team member modal",
    examples: [
      "invite someone to my team",
      "add a new user",
      "how do I invite people?",
    ],
    execute: ({ setShowInviteModal }) => setShowInviteModal(true),
  });

  usePillarTool({
    name: "view_billing",
    type: "navigate",
    description: "Navigate to billing and subscription settings",
    autoRun: true,
    execute: () => router.push("/settings/billing"),
  });
}
