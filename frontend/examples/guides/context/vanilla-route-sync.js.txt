// Sync context on every navigation (vanilla JS SPA)
const pillar = Pillar.getInstance();

// Track the current path
let lastPath = window.location.pathname;

// Poll for path changes (for SPAs without popstate)
setInterval(() => {
  if (window.location.pathname !== lastPath) {
    lastPath = window.location.pathname;
    pillar?.setContext({ currentPage: lastPath });
  }
}, 300);

// Also listen for popstate (back/forward buttons)
window.addEventListener('popstate', () => {
  pillar?.setContext({ currentPage: window.location.pathname });
});
