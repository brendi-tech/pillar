import { Component, Input } from '@angular/core';
import { injectPillarTool, type ToolCardContext } from '@pillar-ai/angular';

interface DeleteData {
  itemId: string;
  itemName: string;
}

@Component({
  selector: 'app-confirm-delete-card',
  standalone: true,
  template: `
    <div class="p-4 border border-red-200 rounded-lg bg-red-50">
      <h3 class="font-semibold text-red-900">Delete {{ data.itemName }}?</h3>
      <p class="text-red-700 text-sm mt-1">This action cannot be undone.</p>
      <div class="flex gap-2 mt-4">
        <button (click)="handleCancel()" [disabled]="loading || !context.isReady">Cancel</button>
        <button
          (click)="handleDelete()"
          [disabled]="loading || !context.isReady"
          class="bg-red-600 text-white"
        >
          {{ loading ? 'Deleting...' : 'Delete' }}
        </button>
      </div>
    </div>
  `,
})
export class ConfirmDeleteCardComponent {
  @Input() data!: DeleteData;
  @Input() sendResult!: (result: Record<string, unknown>) => Promise<void>;
  @Input() context!: ToolCardContext;

  loading = false;

  handleCancel() {
    this.sendResult({ deleted: false });
  }

  async handleDelete() {

      await deleteItem(this.data.itemId);

    } catch {

    }

}

@Component({
  selector: 'app-delete-tools',
  standalone: true,
  template: '',
})
export class DeleteToolsComponent {
  constructor() {
    injectPillarTool({
      name: 'delete_item',
      type: 'inline_ui',
      description: 'Delete an item permanently',
      inputSchema: {
        type: 'object',
        properties: {
          itemId: { type: 'string', description: 'ID of the item to delete' },
          itemName: { type: 'string', description: 'Display name of the item' },
        },
        required: ['itemId', 'itemName'],
      },
      render: ConfirmDeleteCardComponent,
    });
  }
}
