import { usePillarAction, usePillar } from "@pillar-ai/react";

export function useOnboardingAction() {
  const { completeAction } = usePillar();

  usePillarAction({
    name: "start_onboarding",
    type: "trigger_action",
    description: "Start the onboarding wizard",
    execute: () => {
      openOnboardingWizard({
        onComplete: () => {
          // Signal to Pillar that the action is done
          completeAction();
        },
      });
      // Don't return success yet - wizard handles completion
    },
  });
}
