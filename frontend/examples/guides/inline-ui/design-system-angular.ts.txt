import { Component, Input, Output, EventEmitter } from '@angular/core';
import { ButtonComponent, CardComponent, CardHeaderComponent, CardContentComponent } from '@/components/ui';

interface ConfirmData {
  message: string;
}

@Component({
  selector: 'app-confirm-card',
  standalone: true,
  imports: [ButtonComponent, CardComponent, CardHeaderComponent, CardContentComponent],
  template: `
    <app-card>
      <app-card-header>Confirm Action</app-card-header>
      <app-card-content>
        <p>{{ data.message }}</p>
        <div class="flex gap-2 mt-4">
          <app-button variant="outline" (click)="onCancel.emit()">Cancel</app-button>
          <app-button (click)="onConfirm.emit()">Confirm</app-button>
        </div>
      </app-card-content>
    </app-card>
  `,
})
export class ConfirmCardComponent {
  @Input() data!: ConfirmData;
  @Output() onConfirm = new EventEmitter<void>();
  @Output() onCancel = new EventEmitter<void>();
}
